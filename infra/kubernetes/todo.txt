# Déploiement LionWiki-t2t sur Kubernetes
# ==========================================

# 1. (Optionnel) Builder et pousser l'image Docker
#    (sinon l'image publique farvardin4/lionwiki:latest est utilisée)
cd infra/docker/lionwiki
docker build -t monrepo/lionwiki:latest .
docker push monrepo/lionwiki:latest

# Réserver instance CPU sur scaleway (ex DEV1-M mini 4 Go RAM, block storage 10 Go)

# 2. Modifier lionwiki-ingress.yaml : remplacer "yourdomain.org" par votre domaine

# 3. Appliquer les manifests dans l'ordre
kubectl apply -f data-persistentvolumeclaim.yaml
kubectl apply -f lionwiki-deployment.yaml
kubectl apply -f lionwiki-service.yaml
kubectl apply -f lionwiki-ingress.yaml

# 4. Vérifier le déploiement
kubectl get pods -l app=lionwiki
kubectl logs -l app=lionwiki --follow
kubectl get pvc lionwiki-data

# 5. TLS avec cert-manager
#    a) Installer cert-manager
kubectl apply -f https://github.com/cert-manager/cert-manager/releases/latest/download/cert-manager.yaml
#    Attendre que les pods cert-manager soient Ready (~30s)
kubectl wait --namespace cert-manager --for=condition=Ready pod --all --timeout=120s

#    b) Tester d'abord avec le staging (quota illimité, certificat non reconnu par les navigateurs)
#       Modifier letsencrypt-clusterissuer.yaml : changer letsencrypt-prod en letsencrypt-staging dans l'ingress
#       puis passer en prod une fois validé
kubectl apply -f letsencrypt-clusterissuer.yaml

#    c) Appliquer l'ingress avec TLS (yourdomain.org doit pointer vers le cluster)
kubectl apply -f lionwiki-ingress.yaml

#    d) Vérifier l'émission du certificat
kubectl get certificate
kubectl describe certificate lionwiki-tls
#    Le champ "Ready: True" confirme que Let's Encrypt a signé le certificat.
#    En cas d'erreur, inspecter la CertificateRequest et le Challenge :
kubectl get certificaterequest
kubectl describe challenge

# Nettoyage
kubectl delete -f lionwiki-ingress.yaml -f lionwiki-service.yaml -f lionwiki-deployment.yaml -f data-persistentvolumeclaim.yaml
